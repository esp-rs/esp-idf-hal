3// When compiling a binary Rust crate in STD (e.g. in Cargo-first builds) and NOT doing
4// any tricks like using #[no_main] or #[start], the Rust compiler will autogenerate
5// a C function with the signature as below which will be proxying
6// the real Rust main function of your binary crate
7//
8// So to bridge this function with the real C "app_main()" entrypoint
9// that ESP-IDF expects it is enough to implement app_main() and call in it
10// the "main" C function autogenerated by the Rust compiler
11//
12// See https://github.com/rust-lang/rust/issues/29633 for more information
13#[cfg(all(feature = "std", feature = "binstart"))]
14extern "C" {
15    fn main(p1: isize, p2: *const *const u8) -> isize;
16}
17
18// When compiling a static Rust library crate (e.g. by using a PIO->Cargo or a CMake->Cargo) build,
19// there is no main function that the Rust compiler expects, nor autogeneration of a callable
20// wrapper around it.
21//
22// In that case (and if the "libstart" feature is enabled), it is _us_ (not the Rust compiler)
23// expecting the user to define a rust #[no_mangle] "main" function and it is our code below which is explicitly
24// calling it from app_main(). If the user does not define a main() runction in Rust, there will
25// be a linkage error instead of the nice Rust syntax error for binary crates.
26//
27// Another restriction of the "libstart" feature is that the Rust main function will always have one
28// fixed signature: "fn main() -> ()" - as opposed to the flexibility of main() in binary crates
29// where it can have quite a few different returning types
30//
31// When compiling a binary Rust crate in no_std, we end up in identical situation:
32// - There is no Rust "lang = start" item defined
33// - As such, there is no magic C "main" function defined for, which would proxy our binary crate main
34#[cfg(any(all(not(feature = "std"), feature = "binstart"), feature = "libstart"))]
35extern "Rust" {
36    fn main();
37}

39#[no_mangle]
40pub extern "C" fn app_main() {
41    unsafe {
42        // #[cfg(all(feature = "uart0_driver_init", esp_idf_comp_driver_enabled, esp_idf_comp_vfs_enabled))]
43        // uart_init::init_uart0();
44
45        #[cfg(all(feature = "std", feature = "binstart"))]
46        main(0, &[core::ptr::null()] as *const *const u8);
47
48        #[cfg(any(all(not(feature = "std"), feature = "binstart"), feature = "libstart"))]
49        main();
50    }
51}
